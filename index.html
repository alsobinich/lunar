    }

    }

    .events ul {

<body>

  <h1>Подсчет баллов</h1>


  <div id="activeProfessions"></div>


  <div class="profession-selection">

    <h2>Выберите активные профессии:</h2>

    <div id="professionCheckboxes"></div>

  </div>


  <script>

    const professions = [

      { id: 1, name: 'Инженер', icon: '🔧', platinum: 0, aluminum: 0, events: [] },

      { id: 2, name: 'Медик', icon: '⚕️', platinum: 0, aluminum: 0, events: [] },

      { id: 3, name: 'Астронавт', icon: '🚀', platinum: 0, aluminum: 0, events: [] },

      { id: 4, name: 'Строитель', icon: '🏗️', platinum: 0, aluminum: 0, events: [] },

      { id: 5, name: 'Геолог', icon: '🪨', platinum: 0, aluminum: 0, events: [] },

      { id: 6, name: 'Физик', icon: '🔬', platinum: 0, aluminum: 0, events: [] },

      { id: 7, name: 'Эколог', icon: '🌱', platinum: 0, aluminum: 0, events: [] },

      { id: 8, name: 'Программист', icon: '💻', platinum: 0, aluminum: 0, events: [] },

      { id: 9, name: 'Химик', icon: '⚗️', platinum: 0, aluminum: 0, events: [] },

      { id: 10, name: 'Арт-дизайнер', icon: '🎨', platinum: 0, aluminum: 0, events: [] },

    ];


    const activeProfessionsContainer = document.getElementById('activeProfessions');

    const professionCheckboxesContainer = document.getElementById('professionCheckboxes');


    // Render checkboxes

    professions.forEach(profession => {

      const checkbox = document.createElement('input');

      checkbox.type = 'checkbox';

      checkbox.id = `profession-${profession.id}`;

      checkbox.addEventListener('change', () => toggleProfession(profession));


      const label = document.createElement('label');

      label.htmlFor = checkbox.id;

      label.textContent = profession.name;


      const container = document.createElement('div');

      container.appendChild(checkbox);

      container.appendChild(label);


      professionCheckboxesContainer.appendChild(container);

    });


    function toggleProfession(profession) {

      profession.active = !profession.active;

      renderActiveProfessions();

    }


    function addEvent(profession, type, reason) {

      const event = {

        type,

        reason,

        date: new Date().toLocaleString(),

      };

      if (type === 'platinum') profession.platinum++;

      if (type === 'aluminum') profession.aluminum++;

      profession.events.push(event);

      renderActiveProfessions();

    }


    function renderActiveProfessions() {

      activeProfessionsContainer.innerHTML = '';


      professions

        .filter(profession => profession.active)

        .forEach(profession => {

          const container = document.createElement('div');

          container.className = 'profession';


          const icon = document.createElement('div');

          icon.textContent = profession.icon;


          const name = document.createElement('div');

          name.textContent = profession.name;


          const points = document.createElement('div');

          points.className = 'points';

          points.innerHTML = `Платина: ${profession.platinum}<br>Алюминий: ${profession.aluminum}`;


          const events = document.createElement('div');

          events.className = 'events';

          const eventList = document.createElement('ul');

          profession.events.forEach(event => {

            const listItem = document.createElement('li');

            listItem.textContent = `${event.date}: ${event.reason} (${event.type})`;

            eventList.appendChild(listItem);

          });

          events.appendChild(eventList);


          const platinumButton = document.createElement('button');

          platinumButton.textContent = 'Добавить Платину';

          platinumButton.addEventListener('click', () =>

            addEvent(profession, 'platinum', 'За достижение')

          );


          const aluminumButton = document.createElement('button');

          aluminumButton.textContent = 'Добавить Алюминий';

          aluminumButton.addEventListener('click', () =>

            addEvent(profession, 'aluminum', 'За участие')

          );


          container.appendChild(icon);

          container.appendChild(name);

          container.appendChild(points);

          container.appendChild(events);

          container.appendChild(platinumButton);

          container.appendChild(aluminumButton);


          activeProfessionsContainer.appendChild(container);

        });

    }


    renderActiveProfessions();

  </script>

</body>
